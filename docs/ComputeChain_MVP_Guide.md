from typing import List, Optional

    """–û—Å–Ω–æ–≤–Ω–æ–π –∫–ª–∞—Å—Å –±–ª–æ–∫—á–µ–π–Ω–∞"""
    
    def __init__(self, genesis_block: Optional[Block] = None):

        self.mempool = []
        self.state = AccountState()
    
    def _create_genesis(self) -> Block:
        """–°–æ–∑–¥–∞—Ç—å genesis –±–ª–æ–∫"""

            height=0,
            prev_hash="0" * 64,
            timestamp=1700000000,
            validator_address="genesis",
            state_root="0" * 64,
            tx_root="0" * 64
        )
        
        return Block(
            header=header,
            transactions=[],
            signature="genesis_signature"
        )
    
    def add_block(self, block: Block) -> bool:
        """–î–æ–±–∞–≤–∏—Ç—å –±–ª–æ–∫ –≤ —Ü–µ–ø—å"""
        # –í–∞–ª–∏–¥–∏—Ä–æ–≤–∞—Ç—å –±–ª–æ–∫

            return False

        # –û–±–Ω–æ–≤–∏—Ç—å —Å–æ—Å—Ç–æ—è–Ω–∏–µ
        self.state.apply_block(block)
        
        return True

        """–í–∞–ª–∏–¥–∏—Ä–æ–≤–∞—Ç—å –±–ª–æ–∫ –ø–µ—Ä–µ–¥ –¥–æ–±–∞–≤–ª–µ–Ω–∏–µ–º"""

            return False

            return False

        if not block.validate():
            return False
        
        return True

        """–ü–æ–ª—É—á–∏—Ç—å –±–ª–æ–∫ –ø–æ –≤—ã—Å–æ—Ç–µ"""
        if 0 <= height < len(self.chain):
            return self.chain[height]

# blockchain/core/state.py

from typing import Dict
from decimal import Decimal

class Account:
    """–û–±—ä–µ–∫—Ç –∞–∫–∫–∞—É–Ω—Ç–∞"""
    def __init__(self, address: str):
        self.address = address
        self.balance = 0
        self.nonce = 0

class AccountState:
    """–ì–ª–æ–±–∞–ª—å–Ω–æ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ –∞–∫–∫–∞—É–Ω—Ç–æ–≤"""
    
    def __init__(self):
        self.accounts: Dict[str, Account] = {}
    
    def get_balance(self, address: str) -> int:
        """–ü–æ–ª—É—á–∏—Ç—å –±–∞–ª–∞–Ω—Å –∞–∫–∫–∞—É–Ω—Ç–∞"""
        if address not in self.accounts:
            return 0
        return self.accounts[address].balance
    
    def get_nonce(self, address: str) -> int:
        """–ü–æ–ª—É—á–∏—Ç—å nonce –∞–∫–∫–∞—É–Ω—Ç–∞"""
        if address not in self.accounts:
            return 0
        return self.accounts[address].nonce
    
    def transfer(self, from_addr: str, to_addr: str, amount: int) -> bool:
        """–í—ã–ø–æ–ª–Ω–∏—Ç—å –ø–µ—Ä–µ–≤–æ–¥"""
        # –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –±–∞–ª–∞–Ω—Å
        if self.get_balance(from_addr) < amount:
            return False
        
        # –í—ã—á–µ—Å—Ç—å —É –æ—Ç–ø—Ä–∞–≤–∏—Ç–µ–ª—è
        if from_addr not in self.accounts:
            self.accounts[from_addr] = Account(from_addr)
        self.accounts[from_addr].balance -= amount
        self.accounts[from_addr].nonce += 1
        
        # –î–æ–±–∞–≤–∏—Ç—å –ø–æ–ª—É—á–∞—Ç–µ–ª—é
        if to_addr not in self.accounts:
            self.accounts[to_addr] = Account(to_addr)
        self.accounts[to_addr].balance += amount
        
        return True
    
    def mint(self, address: str, amount: int):
        """–°–æ–∑–¥–∞—Ç—å –Ω–æ–≤—ã–µ —Ç–æ–∫–µ–Ω—ã (–Ω–∞–≥—Ä–∞–¥–∞ –∑–∞ –±–ª–æ–∫)"""
        if address not in self.accounts:
            self.accounts[address] = Account(address)
        self.accounts[address].balance += amount
    
    def apply_block(self, block):
        """–ü—Ä–∏–º–µ–Ω–∏—Ç—å —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏–∏ –±–ª–æ–∫–∞ –∫ —Å–æ—Å—Ç–æ—è–Ω–∏—é"""
        for tx in block.transactions:
            if tx['type'] == 'transfer':
                self.transfer(
                    tx['from'],
                    tx['to'],
                    tx['amount']
                )

from typing import List

class ProofOfAuthority:
    """–ü—Ä–æ—Å—Ç–æ–π PoA –∫–æ–Ω—Å–µ–Ω—Å—É—Å"""
    
    def __init__(self, validators: List[str]):

        """Round-robin –≤—ã–±–æ—Ä –≤–∞–ª–∏–¥–∞—Ç–æ—Ä–∞"""

    def is_valid_producer(self, block_height: int, validator_addr: str) -> bool:
        """–ü—Ä–æ–≤–µ—Ä–∏—Ç—å –º–æ–∂–µ—Ç –ª–∏ –≤–∞–ª–∏–¥–∞—Ç–æ—Ä –ø—Ä–æ–∏–∑–≤–µ—Å—Ç–∏ –±–ª–æ–∫"""
