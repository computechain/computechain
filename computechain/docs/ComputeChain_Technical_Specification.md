```python
class MerkleTree:
    """–ë–∏–Ω–∞—Ä–Ω–æ–µ Merkle –¥–µ—Ä–µ–≤–æ"""
    
    def __init__(self):
        self.leaves = []
        self.layers = []
    
    def add_leaf(self, data: bytes):
        """–î–æ–±–∞–≤–∏—Ç—å –ª–∏—Å—Ç"""
        self.leaves.append(sha256(data))
    
    def build(self):
        """–ü–æ—Å—Ç—Ä–æ–∏—Ç—å –¥–µ—Ä–µ–≤–æ"""
        current_layer = self.leaves
        self.layers = [current_layer]
        
        while len(current_layer) > 1:
            next_layer = []
            for i in range(0, len(current_layer), 2):
                left = current_layer[i]
                right = current_layer[i+1] if i+1 < len(current_layer) else left
                parent = sha256(left + right)
                next_layer.append(parent)
            
            self.layers.append(next_layer)
            current_layer = next_layer
    
    def get_root(self) -> bytes32:
        """–ü–æ–ª—É—á–∏—Ç—å Merkle –∫–æ—Ä–µ–Ω—å"""
        return self.layers[-1][0]
    
    def get_proof(self, leaf_index: int) -> List[bytes32]:
        """–°–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞—Ç—å Merkle proof"""
        proof = []
        index = leaf_index
        
        for layer in self.layers[:-1]:
            sibling_index = index ^ 1  # XOR –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è —Å–æ—Å–µ–¥–∞
            if sibling_index < len(layer):
                proof.append(layer[sibling_index])
            index //= 2
        
        return proof

def verify_merkle_proof(leaf: bytes32, index: int, root: bytes32, proof: List[bytes32]) -> bool:
    """–ü—Ä–æ–≤–µ—Ä–∏—Ç—å Merkle proof"""
    current = leaf
    
    for sibling in proof:
        if index % 2 == 0:
            current = sha256(current + sibling)
        else:
            current = sha256(sibling + current)
        index //= 2
    
    return current == root
```
